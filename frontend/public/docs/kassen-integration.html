🏪 BidBlitz Kassen-Integration

Technische Dokumentation für die Integration von BidBlitz Pay in Ihr Kassensystem

Übersicht: Diese Dokumentation erklärt, wie Sie BidBlitz-Zahlungen in Ihr Kassensystem integrieren können. Kunden können mit ihrem BidBlitz-Guthaben an Ihrer Kasse bezahlen - schnell, sicher und ohne Bargeld.

📱 QR-Code Format

Standard-Format (Version 2.0)

Der QR-Code enthält einen JSON-String mit allen notwendigen Informationen:

{
  "type": "bidblitz_pay",
  "version": "2.0",
  "token": "cpt_abc123def456...",
  "customer_id": "usr_xyz789",
  "customer_number": "BID-123456",
  "expires": "2026-02-20T12:30:00Z"
}

Kompakt-Format (für einfache Scanner)

Alternativ ein einfacher String für Scanner die nur Text ausgeben:

BIDBLITZ:2.0:cpt_abc123def456:BID-123456:1708430400

Format: BIDBLITZ:{version}:{token}:{customer_number}:{unix_timestamp}

🔌 API-Endpoints

EndpointMethodeBeschreibung
/api/digital/scan-pay POST Zahlung durchführen (Kasse scannt Kunden-QR)
/api/digital/customer/lookup GET Kunden-Info abrufen (Name, Guthaben prüfen)
/api/digital/payments/{id} GET Zahlungsstatus abfragen

💻 Integrations-Beispiele

Python Integration

import requests
import json

class BidBlitzPOS:
    """BidBlitz Kassen-Integration für Python"""
    
    def __init__(self, api_key: str, api_url: str = "https://bidblitz.ae"):
        self.api_key = api_key
        self.api_url = api_url
        self.headers = {
            "X-API-Key": api_key,
            "Content-Type": "application/json"
        }
    
    def parse_qr_code(self, qr_data: str) -> dict:
        """
        Parst den gescannten QR-Code.
        Unterstützt JSON und Kompakt-Format.
        """
        # JSON-Format
        if qr_data.startswith("{"):
            return json.loads(qr_data)
        
        # Kompakt-Format: BIDBLITZ:2.0:token:customer:timestamp
        if qr_data.startswith("BIDBLITZ:"):
            parts = qr_data.split(":")
            return {
                "type": "bidblitz_pay",
                "version": parts[1],
                "token": parts[2],
                "customer_number": parts[3],
                "expires_timestamp": int(parts[4])
            }
        
        raise ValueError("Ungültiges QR-Code Format")
    
    def process_payment(self, qr_data: str, amount: float) -> dict:
        """
        Führt eine Zahlung durch.
        
        Args:
            qr_data: Der gescannte QR-Code Inhalt
            amount: Betrag in EUR
            
        Returns:
            dict mit Zahlungsergebnis
        """
        qr_info = self.parse_qr_code(qr_data)
        
        response = requests.post(
            f"{self.api_url}/api/digital/scan-pay",
            headers=self.headers,
            json={
                "payment_token": qr_info["token"],
                "amount": amount,
                "customer_number": qr_info.get("customer_number")
            }
        )
        
        result = response.json()
        
        if response.status_code == 200:
            return {
                "success": True,
                "payment_id": result.get("payment_id"),
                "message": f"Zahlung erfolgreich: €{amount:.2f}",
                "customer_name": result.get("customer_name"),
                "new_balance": result.get("new_balance")
            }
        else:
            return {
                "success": False,
                "error": result.get("detail", "Unbekannter Fehler"),
                "code": response.status_code
            }
    
    def lookup_customer(self, customer_number: str) -> dict:
        """Prüft Kundeninformationen vor der Zahlung."""
        response = requests.get(
            f"{self.api_url}/api/digital/customer/lookup",
            headers=self.headers,
            params={"customer_number": customer_number}
        )
        return response.json()

C# / .NET Integration

using System;
using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;

public class BidBlitzPOS
{
    private readonly HttpClient _client;
    private readonly string _apiUrl;

    public BidBlitzPOS(string apiKey, string apiUrl = "https://bidblitz.ae")
    {
        _apiUrl = apiUrl;
        _client = new HttpClient();
        _client.DefaultRequestHeaders.Add("X-API-Key", apiKey);
    }

    public class QRCodeData
    {
        public string Type { get; set; }
        public string Version { get; set; }
        public string Token { get; set; }
        public string CustomerNumber { get; set; }
    }

    public QRCodeData ParseQRCode(string qrData)
    {
        // JSON-Format
        if (qrData.StartsWith("{"))
        {
            return JsonSerializer.Deserialize<QRCodeData>(qrData);
        }
        
        // Kompakt-Format
        if (qrData.StartsWith("BIDBLITZ:"))
        {
            var parts = qrData.Split(':');
            return new QRCodeData
            {
                Type = "bidblitz_pay",
                Version = parts[1],
                Token = parts[2],
                CustomerNumber = parts[3]
            };
        }
        
        throw new ArgumentException("Ungültiges QR-Code Format");
    }

    public async Task<PaymentResult> ProcessPaymentAsync(string qrData, decimal amount)
    {
        var qrInfo = ParseQRCode(qrData);
        
        var payload = new
        {
            payment_token = qrInfo.Token,
            amount = amount,
            customer_number = qrInfo.CustomerNumber
        };
        
        var content = new StringContent(
            JsonSerializer.Serialize(payload),
            Encoding.UTF8,
            "application/json"
        );
        
        var response = await _client.PostAsync(
            $"{_apiUrl}/api/digital/scan-pay",
            content
        );
        
        var json = await response.Content.ReadAsStringAsync();
        var result = JsonSerializer.Deserialize<Dictionary<string, object>>(json);
        
        return new PaymentResult
        {
            Success = response.IsSuccessStatusCode,
            PaymentId = result.GetValueOrDefault("payment_id")?.ToString(),
            Message = response.IsSuccessStatusCode 
                ? $"Zahlung erfolgreich: €{amount:F2}"
                : result.GetValueOrDefault("detail")?.ToString()
        };
    }
}

public class PaymentResult
{
    public bool Success { get; set; }
    public string PaymentId { get; set; }
    public string Message { get; set; }
}

Java Integration

import java.net.http.*;
import java.net.URI;
import com.google.gson.*;

public class BidBlitzPOS {
    private final String apiKey;
    private final String apiUrl;
    private final HttpClient client;
    private final Gson gson;

    public BidBlitzPOS(String apiKey, String apiUrl) {
        this.apiKey = apiKey;
        this.apiUrl = apiUrl;
        this.client = HttpClient.newHttpClient();
        this.gson = new Gson();
    }

    public record QRCodeData(String token, String customerNumber) {}
    
    public record PaymentResult(boolean success, String paymentId, String message) {}

    public QRCodeData parseQRCode(String qrData) {
        if (qrData.startsWith("BIDBLITZ:")) {
            String[] parts = qrData.split(":");
            return new QRCodeData(parts[2], parts[3]);
        }
        // JSON-Format
        JsonObject json = gson.fromJson(qrData, JsonObject.class);
        return new QRCodeData(
            json.get("token").getAsString(),
            json.get("customer_number").getAsString()
        );
    }

    public PaymentResult processPayment(String qrData, double amount) throws Exception {
        QRCodeData qr = parseQRCode(qrData);
        
        JsonObject payload = new JsonObject();
        payload.addProperty("payment_token", qr.token());
        payload.addProperty("amount", amount);
        payload.addProperty("customer_number", qr.customerNumber());
        
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(apiUrl + "/api/digital/scan-pay"))
            .header("X-API-Key", apiKey)
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(gson.toJson(payload)))
            .build();
        
        HttpResponse<String> response = client.send(request, 
            HttpResponse.BodyHandlers.ofString());
        
        JsonObject result = gson.fromJson(response.body(), JsonObject.class);
        
        if (response.statusCode() == 200) {
            return new PaymentResult(true, 
                result.get("payment_id").getAsString(),
                String.format("Zahlung erfolgreich: €%.2f", amount));
        }
        return new PaymentResult(false, null, 
            result.get("detail").getAsString());
    }
}

PHP Integration

<?php

class BidBlitzPOS {
    private string $apiKey;
    private string $apiUrl;
    
    public function __construct(string $apiKey, string $apiUrl = "https://bidblitz.ae") {
        $this->apiKey = $apiKey;
        $this->apiUrl = $apiUrl;
    }
    
    public function parseQRCode(string $qrData): array {
        // JSON-Format
        if (str_starts_with($qrData, '{')) {
            return json_decode($qrData, true);
        }
        
        // Kompakt-Format: BIDBLITZ:2.0:token:customer:timestamp
        if (str_starts_with($qrData, 'BIDBLITZ:')) {
            $parts = explode(':', $qrData);
            return [
                'type' => 'bidblitz_pay',
                'version' => $parts[1],
                'token' => $parts[2],
                'customer_number' => $parts[3],
                'expires_timestamp' => (int)$parts[4]
            ];
        }
        
        throw new Exception('Ungültiges QR-Code Format');
    }
    
    public function processPayment(string $qrData, float $amount): array {
        $qrInfo = $this->parseQRCode($qrData);
        
        $ch = curl_init();
        curl_setopt_array($ch, [
            CURLOPT_URL => $this->apiUrl . '/api/digital/scan-pay',
            CURLOPT_POST => true,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_HTTPHEADER => [
                'X-API-Key: ' . $this->apiKey,
                'Content-Type: application/json'
            ],
            CURLOPT_POSTFIELDS => json_encode([
                'payment_token' => $qrInfo['token'],
                'amount' => $amount,
                'customer_number' => $qrInfo['customer_number'] ?? null
            ])
        ]);
        
        $response = curl_exec($ch);
        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        
        $result = json_decode($response, true);
        
        if ($httpCode === 200) {
            return [
                'success' => true,
                'payment_id' => $result['payment_id'],
                'message' => sprintf('Zahlung erfolgreich: €%.2f', $amount),
                'customer_name' => $result['customer_name'] ?? null,
                'new_balance' => $result['new_balance'] ?? null
            ];
        }
        
        return [
            'success' => false,
            'error' => $result['detail'] ?? 'Unbekannter Fehler',
            'code' => $httpCode
        ];
    }
}

cURL / Shell Integration

#!/bin/bash
# BidBlitz POS - Shell Integration

API_KEY="YOUR_API_KEY_HERE"
API_URL="https://bidblitz.ae"

# Funktion: QR-Code parsen (Kompakt-Format)
parse_qr() {
    local qr="$1"
    if [[ "$qr" == BIDBLITZ:* ]]; then
        IFS=':' read -ra PARTS <<< "$qr"
        TOKEN="${PARTS[2]}"
        CUSTOMER="${PARTS[3]}"
    fi
}

# Funktion: Zahlung durchführen
process_payment() {
    local qr_data="$1"
    local amount="$2"
    
    parse_qr "$qr_data"
    
    RESPONSE=$(curl -s -X POST "$API_URL/api/digital/scan-pay" \
        -H "X-API-Key: $API_KEY" \
        -H "Content-Type: application/json" \
        -d "{
            \"payment_token\": \"$TOKEN\",
            \"amount\": $amount,
            \"customer_number\": \"$CUSTOMER\"
        }")
    
    # Ergebnis prüfen
    if echo "$RESPONSE" | grep -q '"payment_id"'; then
        echo "✅ Zahlung erfolgreich!"
        echo "$RESPONSE" | python3 -c "
import sys, json
d = json.load(sys.stdin)
print(f'   Payment ID: {d.get(\"payment_id\")}')
print(f'   Kunde: {d.get(\"customer_name\", \"N/A\")}')
print(f'   Neues Guthaben: €{d.get(\"new_balance\", 0):.2f}')
"
    else
        echo "❌ Fehler:"
        echo "$RESPONSE" | python3 -c "
import sys, json
d = json.load(sys.stdin)
print(f'   {d.get(\"detail\", \"Unbekannter Fehler\")}')
"
    fi
}

⚙️ Scanner-Konfiguration

USB-Scanner Einrichtung

  1. Scanner an USB-Port anschließen
  2. Scanner in HID-Modus setzen (meist Standard)
  3. Optional: Präfix/Suffix konfigurieren für eindeutige Erkennung
  4. QR-Code scannen → Text erscheint wie Tastatureingabe
